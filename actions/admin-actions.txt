
"use server";

import { prisma } from "@/lib/prisma";
import { revalidatePath } from "next/cache";
import { auth } from "@clerk/nextjs/server";

export async function createNewCompany(formData: FormData) {
  const { userId } = await auth();

  // 1. Security Check
  const sender = await prisma.user.findUnique({
    where: { clerkId: userId as string },
  });

  if (sender?.role !== "SUPER_ADMIN") {
    return { success: false, error: "Unauthorized" };
  }

  const companyName = formData.get("companyName") as string;
  const adminEmail = formData.get("adminEmail") as string;
  const adminName = formData.get("adminName") as string;

  try {
    // We use a transaction to ensure ALL or NOTHING is created
    const newCompany = await prisma.$transaction(async (tx) => {
      // A. Create the Company and the User
      const company = await tx.company.create({
        data: {
          name: companyName,
          users: {
            create: {
              email: adminEmail,
              name: adminName,
              role: "ADMIN",
              status: "PENDING", // Correctly sets them as pending until Clerk sync
            },
          },
        },
      });

      // B. Create the Default Pipeline (CRITICAL STEP)
      // Without this, the CRM has no columns/stages!
      await tx.leadStatus.createMany({
        data: [
          { companyId: company.id, label: "NEW", color: "#3b82f6", order: 0 },
          {
            companyId: company.id,
            label: "QUALIFIED",
            color: "#8b5cf6",
            order: 1,
          },
          {
            companyId: company.id,
            label: "WON",
            color: "#22c55e",
            order: 2,
            isClosing: true,
            isWon: true,
          },
          {
            companyId: company.id,
            label: "LOST",
            color: "#ef4444",
            order: 3,
            isClosing: true,
            isWon: false,
          },
        ],
      });

      // C. Create the Invitation record (For tracking)
      await tx.companyInvitation.create({
        data: {
          companyId: company.id,
          email: adminEmail,
          name: adminName,
          role: "ADMIN",
          invitedBy: sender!.id,
          status: "PENDING",
        },
      });

      return company;
    });

    revalidatePath("/super-admin"); // Specific path revalidation is better
    return { success: true, id: newCompany.id };
  } catch (error) {
    console.error("Error creating company:", error);
    return { success: false, error: "Company name or Email already exists." };
  }
}
